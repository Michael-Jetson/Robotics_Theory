% reproj_error.tex -- reprojection error sketch (tikz-3dplot).
% Camera centre C, image plane, 3D point X, projection ray, projected xhat
% vs observed x, error e = x - xhat. Standalone; build via xelatex.
\documentclass[border=6pt]{standalone}
\usepackage{tikz}\usepackage{tikz-3dplot}
\usetikzlibrary{calc,arrows.meta}
\begin{document}
\tdplotsetmaincoords{70}{120}
\begin{tikzpicture}[tdplot_main_coords, scale=1.25, >=Stealth, line cap=round]
  \coordinate (C) at (0,0,0);                          % camera centre
  \coordinate (i1) at (2,-1,-.8); \coordinate (i2) at (2,1,-.8);
  \coordinate (i3) at (2,1,.8);   \coordinate (i4) at (2,-1,.8);
  \draw[fill=blue!8] (i1)--(i2)--(i3)--(i4)--cycle;    % image plane
  \node at (2,0,-1.05) {\small image plane};
  \coordinate (X)  at (4.5,1.6,1.2);                   % 3D landmark
  \draw[dashed] (C)--(X);                              % projection ray
  \fill (X) circle (1.6pt) node[right]{$\mathbf{X}$};
  \coordinate (xh) at (2,.71,.53);                     % projected (precomputed on plane)
  \coordinate (xo) at (2,.35,.30);                     % observed
  \fill[red]   (xh) circle (1.4pt) node[above right]{$\hat{\mathbf{x}}=\pi(K,T,\mathbf{X})$};
  \fill[black] (xo) circle (1.4pt) node[below left]{$\mathbf{x}$};
  \draw[->,red,thick] (xh)--(xo) node[midway,right]{$\mathbf{e}=\mathbf{x}-\hat{\mathbf{x}}$};
  \fill (C) circle (1.6pt) node[left]{$C$};
\end{tikzpicture}
\end{document}

\documentclass[lang=cn, 10pt, device=normal]{elegantbook}

%==================== 学习笔记批注（行内小字彩注；不入页边——旁注会溢出版心）====================
% 历史：曾用 todonotes 旁注，长批注溢出页面（作者反馈）→ 改行内。todonotes 仍载备用，但本组宏不再调 \todo。
\usepackage{todonotes}
\DeclareRobustCommand{\pz}[1]{{\footnotesize\color{structurecolor}~[批注：#1]}}   % 行内普通批注（蓝）
\DeclareRobustCommand{\pzr}[1]{{\footnotesize\color{red!70!black}~[重点：#1]}}     % 行内重点/警示（红）
% \rebuilt：重建标记。双形——\rebuilt{X} 标具体公式/项；裸 \rebuilt 标整段。
%   裸用经 \@ifnextchar 探测后随非 { 即走无参分支，不吞下一 token（如 \textbf）。
\makeatletter
\DeclareRobustCommand{\rebuilt}{\@ifnextchar\bgroup\rebuilt@arg\rebuilt@bare}
\newcommand{\rebuilt@arg}[1]{{\footnotesize\color{second}~[重建待核对：#1]}}
\newcommand{\rebuilt@bare}{{\footnotesize\color{second}~[重建待核对]}}
\makeatother
% 中文强调：\emph 在本 CJK 字体下默认回退“楷体”(异体、偏大、非斜体)，不宜做术语强调
% → 统一为黑体(\sffamily=中文黑体)，干净且与 \textbf 粗宋形成两级强调。作者反馈。
\renewcommand{\emph}[1]{{\normalfont\sffamily #1}}
% 生僻字回退：Fandol(Song/Kai)缺个别生僻字(如 劼/啰)→ 用 LXGW Neo ZhiSong(广覆盖 Song 体)补字。
% XeLaTeX 无主块内逐字自动回退(那是 LuaLaTeX)，故用此宏手动包住缺字、可复用：\rare{劼}
\newCJKfontfamily\rarecjkfont{LXGW Neo ZhiSong}
\newcommand{\rare}[1]{{\rarecjkfont #1}}
\usepackage{marginnote}
\usepackage{float}   % [H] 精确就位：小结/参考表不浮动，标题紧贴表格，杜绝漂走留白
\usepackage{longtable}   % 长表跨页：高表(如发展脉络)用 longtable，table[H]+tabular 不跨页会冲出页底
\usepackage{array}       % \newcolumntype 支持
% 表格居中规范(集中定义)：短属性列(年份/Venue/数值)用 c；需换行的短属性用 C{宽}(居中定宽)；
%   长文本列用 p{宽}(左对齐正文)但表头单元用 \multicolumn{1}{c}{列名} 居中。表头行一律居中。
\newcolumntype{C}[1]{>{\centering\arraybackslash}p{#1}}

%==================== 全局版式护栏（字号集中管控，杜绝跨章泄漏）====================
% 背景：某章若在正文顶层裸用 \small/\footnotesize 而未组包/未 \normalsize 复位，
%   该字号会跨 \input 泄漏到其后所有章（作者反馈：dense_mapping 起正文塌成 8pt，
%   根因 = leg_odometry 章尾一处裸 \footnotesize）。字号是全局参数，统一在此设护栏：
%   每个 \chapter 开始前强制复位 \normalsize，使任何遗漏的局部字号最多影响本章、绝不外泄。
\let\rbntOrigChapter\chapter
\renewcommand{\chapter}{\normalsize\rbntOrigChapter}
% 细注语义环境：替代“裸 \noindent\footnotesize 起小字段落”这种易泄漏写法。
%   章内小字说明一律用 \begin{finenote}...\end{finenote}（环境=组，字号自动复位）。
\newenvironment{finenote}{\par\noindent\footnotesize}{\par\normalsize}

% 页眉 / 页边 / 字体：全部沿用官方 ElegantBook 默认（不再用 fancyhdr/geometry 覆盖）。

%==================== 数学 / 绘图 ====================
\usepackage{mathtools}
\usepackage{tikz}
\usetikzlibrary{positioning, calc, arrows.meta, fit, backgrounds}
\newcommand{\SO}{\mathrm{SO}}
\newcommand{\SE}{\mathrm{SE}}
\newcommand{\so}{\mathfrak{so}}
\newcommand{\se}{\mathfrak{se}}
% 字体无关的带圈数字：正文/标题/strong 上下文字体缺 U+2460.. 圈字形，直接 Unicode 会丢字；
% \cnum{12} 用 \textcircled 画圈、数字取当前字体，处处可渲染（全书统一，勿在章内 \newcommand 重定义）。
\newcommand{\cnum}[1]{\textcircled{\raisebox{0pt}{\scriptsize #1}}}
% 行内推导收尾的证毕记号（正文裸用 \qed；proof 环境自带 qed，二者不冲突）。
\providecommand{\qed}{\hfill\ensuremath{\square}}

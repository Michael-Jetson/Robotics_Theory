% rodrigues.tex -- Rodrigues rotation geometric decomposition (tikz-3dplot).
% Standalone; build via xelatex. Embedded in the book with \includegraphics.
\documentclass[border=6pt]{standalone}
\usepackage{tikz}
\usepackage{tikz-3dplot}
\usetikzlibrary{calc,arrows.meta}
\begin{document}
\tdplotsetmaincoords{65}{115}                 % view: elevation, azimuth
\begin{tikzpicture}[tdplot_main_coords, scale=1.4,
    >=Stealth, line cap=round,
    vec/.style={-{Stealth[length=3mm]}, thick}]

  \def\a{2}\def\h{2.2}\def\th{60}             % |v_perp|, |v_par|, angle (deg)
  \coordinate (O)   at (0,0,0);
  \coordinate (par) at (0,0,\h);
  \coordinate (vp)  at (\a,0,0);
  \coordinate (vpr) at ({\a*cos(\th)},{\a*sin(\th)},0);
  \coordinate (v)   at (\a,0,\h);
  \coordinate (vr)  at ({\a*cos(\th)},{\a*sin(\th)},\h);

  \fill[gray!12] (-1,-1,0)--(3.2,-1,0)--(3.2,3,0)--(-1,3,0)--cycle;  % perp plane

  \draw[vec,blue]          (O)--(0,0,\h+1.1) node[above]{$\mathbf{k}$};
  \draw[vec,blue!55!black] (O)--(par) node[left]{$\mathbf{v}_\parallel$};
  \draw[vec,red]           (O)--(vp)  node[below right]{$\mathbf{v}_\perp$};
  \draw[vec,orange]        (O)--(0,\a,0) node[above left]{$\mathbf{k}\times\mathbf{v}_\perp$};
  \draw[vec,black]         (O)--(v)   node[above right]{$\mathbf{v}$};
  \draw[vec,red!70!black]  (O)--(vpr) node[above]{$\mathbf{v}_\perp'$};
  \draw[vec,green!55!black](O)--(vr)  node[above right]{$\mathbf{v}_{\mathrm{rot}}$};

  \draw[dashed] (par)--(v);  \draw[dashed] (par)--(vr);  \draw[dashed] (vpr)--(vr);

  % theta arc in the perpendicular (xy) plane -- version-proof, no \tdplotdrawarc
  \draw[->,thick] plot[variable=\t,domain=0:\th,samples=40]
        ({\a*cos(\t)},{\a*sin(\t)},0);
  \node at ({1.2*\a*cos(\th/2)},{1.2*\a*sin(\th/2)},0) {$\theta$};
\end{tikzpicture}
\end{document}
